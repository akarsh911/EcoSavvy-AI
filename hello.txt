<!DOCTYPE html>
<html>
<head>
    <title>Bing Maps Example</title>
    <meta charset="utf-8" />
    <script type='text/javascript' src='https://www.bing.com/api/maps/mapcontrol?callback=loadMapScenario' async defer></script>
    <script type='text/javascript'>
        function loadMapScenario() {
            var map = new Microsoft.Maps.Map(document.getElementById('myMap'), {
                center: new Microsoft.Maps.Location(20.5937, 78.9629), // Center of India
                zoom: 5 // Zoom level
            });

            // Locations for the 10 states in India
            var locations = [
                { state: 'Kerala', location: new Microsoft.Maps.Location(10.8505, 76.2711) },
                { state: 'West Bengal', location: new Microsoft.Maps.Location(22.9868, 87.8550) },
                { state: 'Punjab', location: new Microsoft.Maps.Location(31.1471, 75.3412) },
                { state: 'Nagaland', location: new Microsoft.Maps.Location(26.1584, 94.5624) },
                { state: 'Delhi', location: new Microsoft.Maps.Location(28.7041, 77.1025) },
                { state: 'Maharashtra', location: new Microsoft.Maps.Location(19.7515, 75.7139) },
                { state: 'Arunachal Pradesh', location: new Microsoft.Maps.Location(28.2180, 94.7278) },
                { state: 'Jharkhand', location: new Microsoft.Maps.Location(23.6102, 85.2799) },
                { state: 'Jammu and Kashmir', location: new Microsoft.Maps.Location(33.7782, 76.5762) },
                { state: 'Madhya Pradesh', location: new Microsoft.Maps.Location(22.9734, 78.6569) },
            ];

            // Add pushpins for each location
            for (var i = 0; i < locations.length; i++) {
                var pin = new Microsoft.Maps.Pushpin(locations[i].location, {
                    title: locations[i].state
                });
                map.entities.push(pin);
            }
        }
    </script>
</head>
